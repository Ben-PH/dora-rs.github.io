# Uninstalling

````mdx-code-block
import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';

<Tabs groupId="current-os" queryString>
  <TabItem value="linux" label="Linux">

```bash
sudo rm $(which dora)
sudo rm $(which dora-coordinator)
sudo rm $(which dora-daemon)
pip uninstall dora-rs ## For Python API
dora --help # This should fail
```

  </TabItem>
  <TabItem value="macos" label="MacOS">

```bash
sudo rm $(which dora)
sudo rm $(which dora-coordinator)
sudo rm $(which dora-daemon)
pip uninstall dora-rs ## For Python API
dora --help # This should fail
```

  </TabItem>
  <TabItem value="windows" label="Windows">
Remove dora, dora-coordinator and dora-daemon binary and run:

```bash
pip uninstall dora-rs
```    

  </TabItem>
</Tabs>
````


## Cargo and Pip

```bash
cargo remove dora-cli
cargo remove dora-coordinator
cargo remove dora-daemon
pip uninstall dora-rs ## For Python API
```
